import parser from 'html-react-parser';
import { useState, useEffect } from 'react';
import axios from 'axios';

export const Analyse = () => {
    const [ html, setHtml ] = useState("<h3>Laster...</h3>");
    useEffect(() => {
        axios.get('/dataset/Portefoljeanalyse_010124.html')
        .then(res => {
            setHtml(res.data);
        })
        .catch(error => { 
            setHtml("Noe gikk galt, vennligst oppdater siden.")
        })
    }, []);
    return(
        parser(html)
    )
}

<Analyse />
